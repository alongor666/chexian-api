## 一、业务背景

### 1.1 核心业务指标

车险业务盈利能力评估需要以下四大核心指标：

| 指标 | 计算公式 | 业务价值 |
|------|----------|----------|
| **满期赔付率** | 已报告赔款 / 满期保费 | 评估承保风险，识别高赔付产品/客户 |
| **费用率** | 费用金额 / 保费 | 评估运营效率，控制管理成本 |
| **变动成本率** | 满期赔付率 + 费用率 | 综合评估业务盈利能力 |
| **已赚保费** | 按时间维度确认的收入 | 精算收入确认，支持按期间分析 |

### 1.2 计算规则详解

#### 满期赔付率计算

```
已报告赔款 = 已结案取已决赔款，未结案取未决准备金（二选一，不相加）

满期保费 = 保费 × (满期天数 / 保险期限天数)
满期天数 = min(统计日 - 起保日期, 保险期限天数)
保险期限天数 = 365天（车险标准一年期）

满期赔付率 = 已报告赔款 / 满期保费
```

**说明**：
- 满期保费体现保单在统计时点已经"赚取"的保费部分
- 对于统计日在保单起保前的情况，满期保费为0
- 对于超过保险期限的保单，满期保费=总保费

#### 已赚保费计算

```
已赚保费 = 保费 × (已赚天数 / 保险期限天数)
已赚天数 = min(max(统计日 - 起保日期, 0), 保险期限天数)
保险期限天数 = 365天
```

**说明**：
- 已赚保费是会计准则要求的收入确认方式
- 支持按任意统计日计算已赚保费
- 支持对比不同统计日的已赚保费差异（增量确认）

### 3.2 SQL查询生成器设计

#### 3.2.1 满期赔付率查询（claims.ts）

```typescript
/**
 * 生成满期赔付率分析SQL
 * @param filters 筛选条件（机构、时间范围、险别等）
 * @param dimension 分析维度（org_name/insurance_type/customer_category）
 * @param statDate 统计日（默认为当前日期）
 */
export function generateLossRatioQuery(
  filters: AdvancedFilterState,
  dimension: 'org_name' | 'insurance_type' | 'customer_category',
  statDate: string = 'CURRENT_DATE'  // DC-002 Exception: 满期保费需要统计日参数
): string {
  return `
    WITH PolicyClaims AS (
      -- 关联保单与赔案数据
      SELECT
        p.policy_no,
        p.${dimension},
        p.premium,
        p.effective_date,
        p.insurance_period,
        COALESCE(SUM(c.settled_amount), 0) AS total_settled,
        COALESCE(SUM(c.outstanding_amount), 0) AS total_outstanding
      FROM PolicyFact p
      LEFT JOIN Claims c ON p.policy_no = c.policy_no
      ${buildWhereClauseFromFilters(filters)}
      GROUP BY p.policy_no, p.${dimension}, p.premium, p.effective_date, p.insurance_period
    ),
    MetricsCalc AS (
      -- 计算满期保费和赔付率
      SELECT
        ${dimension},
        SUM(premium) AS total_premium,
        SUM(CASE WHEN claim_status = '已决' THEN total_settled ELSE total_outstanding END) AS total_reported_claims,
        -- 满期天数 = min(统计日 - 起保日期, 保险期限)
        SUM(
          premium *
          LEAST(
            GREATEST(${statDate} - effective_date, 0),
            insurance_period
          ) / insurance_period
        ) AS total_earned_premium
      FROM PolicyClaims
      GROUP BY ${dimension}
    )
    SELECT
      ${dimension},
      total_premium,
      total_reported_claims,
      total_earned_premium,
      -- 满期赔付率
      CASE
        WHEN total_earned_premium > 0
        THEN total_reported_claims / total_earned_premium
        ELSE 0
      END AS loss_ratio,
      -- 排序用
      total_reported_claims DESC
    FROM MetricsCalc
    ORDER BY total_reported_claims DESC;
  `;
}
```

#### 3.2.2 已赚保费查询（earned-premium.ts）

```typescript
/**
 * 生成已赚保费分析SQL（支持多统计日对比）
 * @param filters 筛选条件
 * @param statDates 统计日数组（如 ['2026-01-01', '2026-06-30', '2026-12-31']）
 * @param dimension 分析维度
 */
export function generateEarnedPremiumQuery(
  filters: AdvancedFilterState,
  statDates: string[],
  dimension: 'org_name' | 'insurance_type'
): string {
  // 生成每个统计日的子查询
  const subQueries = statDates.map((statDate, idx) => `
    SELECT
      ${dimension},
      '${statDate}' AS stat_date,
      SUM(premium) AS total_premium,
      SUM(
        premium *
        LEAST(
          GREATEST(CAST('${statDate}' AS DATE) - effective_date, 0),
          365
        ) / 365.0
      ) AS earned_premium
    FROM PolicyFact
    ${buildWhereClauseFromFilters(filters)}
    GROUP BY ${dimension}
  `);

  return `
    WITH EarnedPremiumByDate AS (
      ${subQueries.join('\n    UNION ALL\n')}
    )
    SELECT
      ${dimension},
      stat_date,
      total_premium,
      earned_premium,
      -- 已赚占比
      CASE
        WHEN total_premium > 0
        THEN earned_premium / total_premium
        ELSE 0
      END AS earned_ratio
    FROM EarnedPremiumByDate
    ORDER BY ${dimension}, stat_date;
  `;
}
```

#### 3.3.2 查询缓存

```typescript
// src/shared/services/QueryCacheService.ts
export class QueryCacheService {
  private cache = new Map<string, { data: any; timestamp: number }>();
  private TTL = 5 * 60 * 1000; // 5分钟缓存

  async query(key: string, queryFn: () => Promise<any>): Promise<any> {
    const cached = this.cache.get(key);
    if (cached && Date.now() - cached.timestamp < this.TTL) {
      return cached.data;
    }

    const data = await queryFn();
    this.cache.set(key, { data, timestamp: Date.now() });
    return data;
  }
}
```

```typescript
/**
 * 瀑布图展示盈利能力分解
 * 保费收入 → 已赚保费 → 扣除赔付 → 扣除费用 → 毛利润
 */
export function ProfitabilityWaterfallChart({ data }: Props) {
  const categories = [
    '保费收入',
    '已赚保费',
    '赔付支出',
    '费用支出',
    '毛利润'
  ];

  const values = [
    data.totalPremium,
    data.earnedPremium - data.totalPremium,
    -data.claimAmount,
    -data.expenseAmount,
    data.earnedPremium - data.claimAmount - data.expenseAmount
  ];

  return <WaterfallChart categories={categories} values={values} />;
}
```
