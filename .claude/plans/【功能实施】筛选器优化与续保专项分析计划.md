```typescript
import React, { useState } from 'react';
import DatePicker from 'react-datepicker';
import 'react-datepicker/dist/react-datepicker.css';

interface DateRangePickerProps {
  startDate?: string;
  endDate?: string;
  onChange: (start: string, end: string) => void;
}

export const DateRangePicker: React.FC<DateRangePickerProps> = ({
  startDate,
  endDate,
  onChange
}) => {
  const [start, setStart] = useState<Date | null>(
    startDate ? new Date(startDate) : null
  );
  const [end, setEnd] = useState<Date | null>(
    endDate ? new Date(endDate) : null
  );

  const handleStartChange = (date: Date | null) => {
    setStart(date);
    if (date && end) {
      onChange(formatDate(date), formatDate(end));
    }
  };

  const handleEndChange = (date: Date | null) => {
    setEnd(date);
    if (start && date) {
      onChange(formatDate(start), formatDate(date));
    }
  };

  const formatDate = (date: Date): string => {
    return date.toISOString().split('T')[0];
  };

  return (
    <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
      <div>
        <label className="block text-sm font-medium text-gray-700 mb-1">
          签单日期起始
        </label>
        <DatePicker
          selected={start}
          onChange={handleStartChange}
          selectsStart
          startDate={start}
          endDate={end}
          placeholderText="选择开始日期"
          className="w-full px-3 py-2 border border-gray-300 rounded text-sm"
          dateFormat="yyyy-MM-dd"
          isClearable
        />
      </div>
      <div>
        <label className="block text-sm font-medium text-gray-700 mb-1">
          签单日期截止
        </label>
        <DatePicker
          selected={end}
          onChange={handleEndChange}
          selectsEnd
          startDate={start}
          endDate={end}
          minDate={start}
          placeholderText="选择截止日期"
          className="w-full px-3 py-2 border border-gray-300 rounded text-sm"
          dateFormat="yyyy-MM-dd"
          isClearable
        />
      </div>
    </div>
  );
};
```

```typescript
const loadOrgSalesmanMapping = async () => {
  const sql = `
    SELECT DISTINCT org_level_3, salesman_name
    FROM PolicyFact
    WHERE org_level_3 IS NOT NULL AND salesman_name IS NOT NULL
    ORDER BY org_level_3, salesman_name
  `;
  const table = await duckdbClient.query(sql).promise;
  const rows = table.toArray();

  const cache: OrgSalesmanCache = {};
  rows.forEach((row: any) => {
    const org = row.org_level_3;
    const salesman = row.salesman_name;
    if (!cache[org]) cache[org] = [];
    if (!cache[org].includes(salesman)) cache[org].push(salesman);
  });

  setOrgSalesmanCache(cache);
};

// 在数据加载后调用
useEffect(() => {
  if (duckdbClient.isDataLoaded() && Object.keys(orgSalesmanCache).length === 0) {
    loadOrgSalesmanMapping();
  }
}, [duckdbClient]);
```

```typescript
const filterSalesmanOptions = (
  options: DimensionOption[],
  searchTerm: string,
  selectedValues: string[],
  availableSalesmen?: string[]  // 新增参数
): DimensionOption[] => {
  const normalizedSearch = searchTerm.trim().toLowerCase();
  const availableSet = new Set(availableSalesmen || []);

  return options.filter(option => {
    const matches = !normalizedSearch || option.value.toLowerCase().includes(normalizedSearch);
    const isAvailable = availableSet.size === 0 || availableSet.has(option.value);
    return (matches || selectedValues.includes(option.value)) && isAvailable;
  });
};
```

**示例：续保率KPI查询**

```typescript
export function generateRenewalKpiQuery(filters: AdvancedFilterState): string {
  const whereClause = buildWhereClauseFromFilters(filters);

  return `
    SELECT
      COUNT(CASE WHEN is_renewal THEN 1 END) * 1.0 / COUNT(*) AS overall_renewal_rate,
      COUNT(CASE WHEN is_renewal THEN 1 END) AS renewal_count,
      COUNT(*) AS total_count
    FROM PolicyFact
    WHERE ${whereClause}
  `;
}
```

**续保单号关联逻辑**（关键需求）：

```typescript
export function generateSuccessfulRenewalQuery(filters: AdvancedFilterState): string {
  const whereClause = buildWhereClauseFromFilters(filters);

  return `
    WITH current_policies AS (
      SELECT
        policy_no,
        renewal_policy_no,
        salesman_name,
        premium
      FROM PolicyFact
      WHERE ${whereClause}
        AND renewal_policy_no IS NOT NULL
    ),
    historical_policies AS (
      SELECT
        policy_no,
        premium
      FROM PolicyFact
      WHERE policy_date < '2025-01-01'  -- 历史保单
    )
    SELECT
      cp.policy_no,
      cp.renewal_policy_no,
      cp.salesman_name,
      cp.premium AS current_premium,
      hp.premium AS historical_premium,
      CASE
        WHEN hp.policy_no IS NOT NULL THEN '成功续保'
        ELSE '续保单号未匹配'
      END AS renewal_status
    FROM current_policies cp
    LEFT JOIN historical_policies hp ON cp.renewal_policy_no = hp.policy_no
  `;
}
```

```typescript
        <table>
          <thead>

              <th>到期日期</th>
              <th>剩余天数</th>
              <th>保费</th>
            </tr>
          </thead>
          <tbody>
            {expiringData.map((row, idx) => (
              <tr key={idx}>
                <td>{row.policy_no}</td>
                <td>{row.salesman_name}</td>
                <td>{row.insurance_end_date}</td>
                <td>{row.days_until_expiry}天</td>

              </tr>
            ))}
          </tbody>
        </table>

    </div>
  );
};
```
