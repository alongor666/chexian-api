# DC-001 规范实施方案：数据分析三要素强制前置

## 执行摘要

本方案详细设计如何为车险数据分析系统实现**数据分析三要素强制前置（DC-001规范）**，包括：

1. **日期口径选择器**（用户可切换签单日期/起保日期）
2. **统一分析年度选择器**（动态生成选项，默认当前年）
3. **动态默认日期范围**（开始日期=当年1月1日，结束日期=数据最大日期）
4. **SQL生成器重构**（支持动态日期字段）

**实施策略**：一次性完整实现（B051-B054全部完成）

```typescript
/**
 * 从高级筛选状态构建WHERE子句（支持动态日期字段）
 *
 * @param filters - 高级筛选状态
 * @param dateField - 日期字段名（可选，默认从 filters.dateCriteria 推断）
 * @returns WHERE子句（不包含WHERE关键字）
 */
export function buildWhereClauseFromFilters(
  filters: AdvancedFilterState,
  dateField?: string
): string {
  const conditions: string[] = ['1=1'];

  // 新增：推断日期字段
  const resolvedDateField = dateField || filters.dateCriteria || 'policy_date';

  // Date range filters（使用动态日期字段）
  if (filters.policy_date_start) {
    conditions.push(`${resolvedDateField} >= '${sanitizeDate(filters.policy_date_start)}'`);
  }
  if (filters.policy_date_end) {
    conditions.push(`${resolvedDateField} <= '${sanitizeDate(filters.policy_date_end)}'`);
  }

  // 其他筛选条件保持不变...
  // Multi-select filters、Boolean filters 等

  return conditions.join(' AND ');
}
```

```typescript
/**
 * 生成保费趋势查询SQL（按机构分组）
 *
 * @param timeView - 时间视图：daily/weekly/monthly
 * @param whereClause - WHERE子句（不包含WHERE关键字）
 * @param dateField - 日期字段名（默认'policy_date'）
 * @returns SQL查询字符串
 */
export function generatePremiumTrendQuery(
  timeView: TimeView,
  whereClause: string = '1=1',
  dateField: string = 'policy_date' // 新增
): string {
  let timeDimension: string;
  let monthKeyDimension: string;
  let nextMonthCondition: string;
  let weekNumberExpression: string;

  switch (timeView) {
    case 'daily':
      // 按日统计
      timeDimension = `CAST(${dateField} AS VARCHAR)`; // 改为动态
      monthKeyDimension = `STRFTIME(${dateField}, '%Y-%m')`; // 改为动态
      // 次月起保：起保日期在签单日期的次月
      // ⚠️ 注意：这里仍然使用 insurance_start_date（业务规则固定）
      nextMonthCondition = `
        (EXTRACT(YEAR FROM insurance_start_date) = EXTRACT(YEAR FROM ${dateField})
         AND EXTRACT(MONTH FROM insurance_start_date) = EXTRACT(MONTH FROM ${dateField}) + 1)
        OR
        (EXTRACT(YEAR FROM insurance_start_date) = EXTRACT(YEAR FROM ${dateField}) + 1
         AND EXTRACT(MONTH FROM ${dateField}) = 12
         AND EXTRACT(MONTH FROM insurance_start_date) = 1)
      `;
      break;

    case 'weekly':
      // 按自然周统计
      weekNumberExpression = `
        CASE
          WHEN DAYOFYEAR(${dateField}) <= (8 - ISODOW(DATE_TRUNC('year', ${dateField})))
          THEN 1
          ELSE CAST(CEIL((DAYOFYEAR(${dateField}) - (8 - ISODOW(DATE_TRUNC('year', ${dateField})))) / 7.0) AS INTEGER) + 1
        END
      `;
      timeDimension = `CONCAT(
        CAST(YEAR(${dateField}) AS VARCHAR),
        '-W',
        LPAD(CAST(${weekNumberExpression} AS VARCHAR), 2, '0')
      )`;
      monthKeyDimension = `
        STRFTIME(
          CASE
            WHEN ${weekNumberExpression} = 1
            THEN CAST(DATE_TRUNC('year', ${dateField}) AS DATE)
            ELSE CAST(DATE_TRUNC('year', ${dateField}) AS DATE) + INTERVAL '1 day' * (
              (8 - ISODOW(DATE_TRUNC('year', ${dateField})))
              + (${weekNumberExpression} - 2) * 7 + 1
            )
          END,
          '%Y-%m'
        )
      `;
      // 次月起保条件（同样使用 insurance_start_date）
      nextMonthCondition = `
        (YEAR(insurance_start_date) = YEAR(${dateField})
         AND MONTH(insurance_start_date) = MONTH(${dateField}) + 1)
        OR
        (YEAR(insurance_start_date) = YEAR(${dateField}) + 1
         AND MONTH(${dateField}) = 12
         AND MONTH(insurance_start_date) = 1)
      `;
      break;

    case 'monthly':
      // 按月统计
      timeDimension = `STRFTIME(${dateField}, '%Y-%m')`; // 改为动态
      monthKeyDimension = `STRFTIME(${dateField}, '%Y-%m')`; // 改为动态
      // 次月起保条件（同样使用 insurance_start_date）
      nextMonthCondition = `
        (YEAR(insurance_start_date) = YEAR(${dateField})
         AND MONTH(insurance_start_date) = MONTH(${dateField}) + 1)
        OR
        (YEAR(insurance_start_date) = YEAR(${dateField}) + 1
         AND MONTH(${dateField}) = 12
         AND MONTH(insurance_start_date) = 1)
      `;
      break;

    default:
      throw new Error(`Unknown time view: ${timeView}`);
  }

  return `
    WITH base_data AS (
      SELECT
        ${timeDimension} AS time_period,
        ${monthKeyDimension} AS month_key,
        org_level_3,
        SUM(premium) AS premium,
        SUM(CASE
          WHEN STRFTIME(${dateField}, '%Y-%m') = ${monthKeyDimension}
          THEN premium
          ELSE 0
        END) AS anchor_month_premium,
        SUM(CASE
          WHEN STRFTIME(${dateField}, '%Y-%m') = ${monthKeyDimension}
               AND (${nextMonthCondition})
          THEN premium
          ELSE 0
        END) AS anchor_month_next_premium
      FROM PolicyFact
      WHERE ${whereClause}
      GROUP BY ${timeDimension}, ${monthKeyDimension}, org_level_3
    ),
    cumulative_stats AS (
      SELECT
        time_period,
        month_key,
        org_level_3,
        premium,
        SUM(anchor_month_premium) OVER (
          PARTITION BY org_level_3, month_key
          ORDER BY time_period
          ROWS UNBOUNDED PRECEDING
        ) AS cumulative_premium,
        SUM(anchor_month_next_premium) OVER (
          PARTITION BY org_level_3, month_key
          ORDER BY time_period
          ROWS UNBOUNDED PRECEDING
        ) AS cumulative_next_month_premium
      FROM base_data
    )
    SELECT
      time_period,
      org_level_3,
      premium,
      0 AS next_month_start_premium,
      CASE
        WHEN cumulative_premium > 0 THEN
          cumulative_next_month_premium / cumulative_premium
        ELSE 0
      END AS next_month_ratio
    FROM cumulative_stats
    ORDER BY time_period, org_level_3
  `;
}

/**
```
```typescript
 * 生成时间周期表达式（支持动态日期字段）
 */
function generateTimeExpression(
  timeView: TimeView, 
  dateColumn: string = 'policy_date' // 新增参数
): string {
  switch (timeView) {
    case 'daily':
      return `CAST(${dateColumn} AS DATE)`;
    case 'weekly':
      return `DATE_TRUNC('week', CAST(${dateColumn} AS DATE))`;
    case 'monthly':
      return `DATE_TRUNC('month', CAST(${dateColumn} AS DATE))`;
    case 'quarterly':
      return `DATE_TRUNC('quarter', CAST(${dateColumn} AS DATE))`;
    case 'yearly':
      return `DATE_TRUNC('year', CAST(${dateColumn} AS DATE))`;
    default:
      throw new Error(`Unknown time view: ${timeView}`);
  }
}

/**
 * 生成同比增长率查询SQL
 * @param config - 增长率配置
 * @param dateField - 日期字段名（新增）
 */
export function generateYoYGrowthQuery(
  config: GrowthConfig,
  dateField?: string // 新增
): string {
  const { 
    timeView, 
    metric = 'SUM(premium)', 
    groupBy = [], 
    whereClause = '1=1' 
  } = config;
  
  const resolvedDateField = dateField || 'policy_date'; // 新增
  const timeExpression = generateTimeExpression(timeView, resolvedDateField); // 传递动态字段
  const groupByClause = groupBy.length > 0 ? `, ${groupBy.join(', ')}` : '';

  return `
    WITH current_period AS (
      SELECT
        ${timeExpression} AS time_period,
        ${metric} AS current_value${groupByClause}
      FROM PolicyFact
      WHERE ${whereClause}
      GROUP BY ${timeExpression}${groupByClause}
    ),
    previous_period AS (
      SELECT
        ${timeExpression} AS time_period,
        ${metric} AS previous_value${groupByClause}
      FROM PolicyFact
      WHERE ${whereClause}
        AND ${timeExpression} >= DATE_SUB(CAST(${timeExpression} AS DATE), INTERVAL '1 year')
        AND ${timeExpression} < DATE_SUB(CAST(${timeExpression} AS DATE), INTERVAL '1 day')
      GROUP BY ${timeExpression}${groupByClause}
    )
    SELECT
      COALESCE(c.time_period, p.time_period) AS time_period,
      COALESCE(c.current_value, 0) AS current_value,
      COALESCE(p.previous_value, 0) AS previous_value,
      CASE
        WHEN COALESCE(p.previous_value, 0) = 0 THEN NULL
        ELSE (COALESCE(c.current_value, 0) - COALESCE(p.previous_value, 0)) / p.previous_value
      END AS growth_rate
      ${groupByClause}
    FROM current_period c
    FULL OUTER JOIN previous_period p ON 
      c.time_period = DATE_ADD(p.time_period, INTERVAL '1 year')
  `;
}
```

```typescript
/**
 * 查询数据中日期字段的最大值
 * 
 * @param dateField - 日期字段名（'policy_date' 或 'insurance_start_date'）
 * @returns 最大日期（YYYY-MM-DD格式）
 */
async getMaxDataDate(dateField: string = 'policy_date'): Promise<string> {
  const sql = `
    SELECT MAX(CAST(${dateField} AS DATE)) AS max_date
    FROM PolicyFact
    WHERE ${dateField} IS NOT NULL
  `;
  
  const table = await this.query(sql).promise;
  const rows = table.toArray();
  const maxDate = rows[0]?.max_date;
  
  if (!maxDate) {
    // 如果数据为空，返回当前日期
    return new Date().toISOString().split('T')[0];
  }
  
  return String(maxDate);
}

/**
 * 查询数据中存在的年份列表
 * 
 * @param dateField - 日期字段名（'policy_date' 或 'insurance_start_date'）
 * @returns 年份数组（降序排列）
 */
async getAvailableYears(dateField: string = 'policy_date'): Promise<number[]> {
  const sql = `
    SELECT DISTINCT YEAR(CAST(${dateField} AS DATE)) AS year
    FROM PolicyFact
    WHERE ${dateField} IS NOT NULL
    ORDER BY year DESC
  `;
  
  const table = await this.query(sql).promise;
  const rows = table.toArray();
  
  return rows.map((row) => Number(row.year));
}
```

```typescript
      setAvailableYears([new Date().getFullYear()]);
    }

  const loadDefaultPolicyMonth = useCallback(async () => {
    try {
      // 如果已有数据元数据，优先使用

        const defaultYear = availableYears.includes(currentYear) 
          ? currentYear 
          : availableYears[0];

          analysisYear: defaultYear,
          dateCriteria: prev.dateCriteria || 'policy_date',
          policy_date_start: `${defaultYear}-01-01`,
          policy_date_end: maxDataDate,

        return;

        FROM PolicyFact
        WHERE policy_date IS NOT NULL
        ORDER BY month_key DESC

      const table = await promise;
      const months = table
        .toArray()
        .map((row: QueryResultRow) => String(row.month_key))
        .filter((monthKey: string | null) => Boolean(monthKey)) as string[];
      
      if (months.length === 0) {
        setFilters((prev) => ({
          ...prev,
          policy_date_start: undefined,
          policy_date_end: undefined,
        }));

      }

      const previousMonth = new Date();
      previousMonth.setMonth(previousMonth.getMonth() - 1);
      const preferredMonth = getMonthKey(previousMonth);

    setFilters,
    filterOptions,
    isFilterCollapsed,
    toggleFilterCollapsed,
    availableSalesmen,
    loadFilterOptions,
    loadDefaultPolicyMonth,
    loadOrgSalesmanMapping,
```

## 附录A：SQL示例

### 场景1：签单日期口径

```sql
-- 用户选择：分析年度=2026，日期口径=签单日期，范围=2026-01-01 ~ 2026-12-31
-- 生成SQL：

SELECT
  STRFTIME(policy_date, '%Y-%m') AS time_period,
  org_level_3,
  SUM(premium) AS premium
FROM PolicyFact
WHERE policy_date >= DATE '2026-01-01'
  AND policy_date <= DATE '2026-12-31'
  AND 1=1
GROUP BY STRFTIME(policy_date, '%Y-%m'), org_level_3
ORDER BY time_period, org_level_3
```

### 场景2：起保日期口径

```sql
-- 用户选择：分析年度=2026，日期口径=起保日期，范围=2026-01-01 ~ 2026-12-31
-- 生成SQL：

SELECT
  STRFTIME(insurance_start_date, '%Y-%m') AS time_period,
  org_level_3,
  SUM(premium) AS premium
FROM PolicyFact
WHERE insurance_start_date >= DATE '2026-01-01'
  AND insurance_start_date <= DATE '2026-12-31'
  AND 1=1
GROUP BY STRFTIME(insurance_start_date, '%Y-%m'), org_level_3
ORDER BY time_period, org_level_3
```
