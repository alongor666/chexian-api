def analyze_periodic_ranking(self):
    """周期性排名分析 - 追踪Top业务员/机构的时间动态"""
    print("【11/14】分析周期性排名变化...")

    result = {
        '业务员排名变化': [],
        '机构排名变化': []
    }

    # 配置参数
    top_n = self.config['business_analysis'].get('top_n', 10)
    time_field = '年月'  # 按月统计

    # 业务员维度
    if '业务员' in self.df.columns and '签单保费' in self.df.columns:
        # 按月分组计算每个业务员的总保费
        monthly_sales = self.df.groupby([time_field, '业务员'])['签单保费'].sum().reset_index()

        # 每个月取Top N
        monthly_top = []
        for month in sorted(monthly_sales[time_field].unique()):
            month_data = monthly_sales[monthly_sales[time_field] == month]
            top_performers = month_data.nlargest(top_n, '签单保费')

            for rank, (_, row) in enumerate(top_performers.iterrows(), 1):
                monthly_top.append({
                    '月份': month,
                    '业务员': row['业务员'],
                    '排名': rank,
                    '保费': float(row['签单保费'])
                })

        result['业务员排名变化'] = monthly_top

    # 机构维度（同样逻辑）
    if '三级机构' in self.df.columns and '签单保费' in self.df.columns:
        monthly_orgs = self.df.groupby([time_field, '三级机构'])['签单保费'].sum().reset_index()

        monthly_top_orgs = []
        for month in sorted(monthly_orgs[time_field].unique()):
            month_data = monthly_orgs[monthly_orgs[time_field] == month]
            top_orgs = month_data.nlargest(top_n, '签单保费')

            for rank, (_, row) in enumerate(top_orgs.iterrows(), 1):
                monthly_top_orgs.append({
                    '月份': month,
                    '机构': row['三级机构'],
                    '排名': rank,
                    '保费': float(row['签单保费'])
                })

        result['机构排名变化'] = monthly_top_orgs

    self.analysis['周期性排名'] = result
    return result
```

/**
 * 周期性排名变化图表（堆叠折线图）

renderPeriodicRankingChart() {
    if (!this.data || this.data.length === 0) return;

    // 按年月分组，计算每个月的Top 10业务员
    const monthlyData = {};

    this.data.forEach(row => {
        const month = row['年月'];
        if (!monthlyData[month]) {
            monthlyData[month] = {};
        }

        const person = row['业务员'];
        if (!monthlyData[month][person]) {
            monthlyData[month][person] = 0;
        }

        monthlyData[month][person] += parseFloat(row['签单保费']) || 0;
    });

    // 提取所有月份并排序
    const months = Object.keys(monthlyData).sort();

    // 收集所有出现过的Top业务员
    const topPerformersSet = new Set();
    months.forEach(month => {
        const sorted = Object.entries(monthlyData[month])
            .sort((a, b) => b[1] - a[1])
            .slice(0, 10);
        sorted.forEach(([person]) => topPerformersSet.add(person));
    });

    const topPerformers = Array.from(topPerformersSet);

    // 构建系列数据
    const series = topPerformers.map(person => {
        return {
            name: person,
            type: 'line',
            data: months.map(month => monthlyData[month][person] || 0),

            symbolSize: 6
        };

def analyze_cross_dimension(self):
    """险别×客户类别交叉分析"""
    print("【12/14】分析险别×客户类别交叉分布...")

    result = {}

    if '险别' in self.df.columns and '客户类别' in self.df.columns and '签单保费' in self.df.columns:
        # 使用 pandas crosstab 进行交叉统计
        cross_table = pd.crosstab(
            index=self.df['客户类别'],
            columns=self.df['险别'],
            values=self.df['签单保费'],
            aggfunc='sum',
            margins=True  # 添加总计行列
        )

        # 转换为JSON友好格式
        result['交叉统计表'] = {
            '客户类别': cross_table.index.tolist(),
            '险别': cross_table.columns.tolist(),
            '数据': cross_table.values.tolist()
        }

        # 计算占比矩阵（按客户类别归一化）
        cross_pct = cross_table.div(cross_table.sum(axis=1), axis=0) * 100
        result['占比矩阵'] = {
            '客户类别': cross_pct.index.tolist(),
            '险别': cross_pct.columns.tolist(),
            '占比': cross_pct.values.tolist()
        }

        # 提取Top组合
        flat_data = []
        for customer in cross_table.index[:-1]:  # 排除总计行
            for insurance in cross_table.columns[:-1]:  # 排除总计列
                premium = cross_table.loc[customer, insurance]
                if pd.notna(premium) and premium > 0:
                    flat_data.append({
                        '客户类别': customer,
                        '险别': insurance,
                        '保费': float(premium)
                    })

        flat_data.sort(key=lambda x: x['保费'], reverse=True)
        result['Top组合'] = flat_data[:20]

    self.analysis['交叉分析'] = result
    return result
```

def analyze_channel_efficiency(self):
    """渠道效率对比分析"""
    print("【13/14】分析渠道效率...")

    result = {
        '渠道效率指标': []
    }

    if '终端来源' in self.df.columns and '签单保费' in self.df.columns and '保单数' in self.df.columns:
        channel_stats = self.df.groupby('终端来源').agg({
            '签单保费': 'sum',
            '保单数': 'sum'
        }).reset_index()

        # 计算衍生指标
        total_premium = channel_stats['签单保费'].sum()

        channel_stats['平均保费'] = channel_stats['签单保费'] / channel_stats['保单数']
        channel_stats['保费占比'] = channel_stats['签单保费'] / total_premium * 100

        # 排序并转换为JSON
        channel_stats = channel_stats.sort_values('签单保费', ascending=False)

        for _, row in channel_stats.iterrows():
            result['渠道效率指标'].append({
                '渠道': row['终端来源'],
                '总保费': float(row['签单保费']),
                '保单数': int(row['保单数']),
                '平均保费': float(row['平均保费']),
                '保费占比': float(row['保费占比'])
            })

    self.analysis['渠道效率'] = result
    return result
```

/**
 * 渠道效率对比图表（雷达图）
 */
renderChannelEfficiencyChart() {
    if (!this.data || this.data.length === 0) return;

    // 聚合渠道数据
    const channelData = {};

    this.data.forEach(row => {
        const channel = row['终端来源'];
        if (!channelData[channel]) {
            channelData[channel] = {
                totalPremium: 0,
                policyCount: 0
            };
        }
        channelData[channel].totalPremium += parseFloat(row['签单保费']) || 0;
        channelData[channel].policyCount += parseInt(row['保单数']) || 0;
    });

    // 计算各渠道指标
    const channels = Object.keys(channelData);
    const totalPremium = Object.values(channelData).reduce((sum, c) => sum + c.totalPremium, 0);

    const indicators = [
        { name: '总保费占比', max: 100 },
        { name: '保单数占比', max: 100 },
        { name: '平均保费', max: 0 },  // 动态计算max
        { name: '渠道效率指数', max: 100 }
    ];

    // 计算平均保费最大值（用于归一化）
    const avgPremiums = channels.map(ch => channelData[ch].totalPremium / channelData[ch].policyCount);
    const maxAvgPremium = Math.max(...avgPremiums);
    indicators[2].max = maxAvgPremium * 1.2;  // 留出20%余量

    const series = channels.map(channel => {
        const data = channelData[channel];
        const avgPremium = data.totalPremium / data.policyCount;
        const premiumShare = (data.totalPremium / totalPremium) * 100;
        const policyShare = (data.policyCount / Object.values(channelData).reduce((sum, c) => sum + c.policyCount, 0)) * 100;

        // 综合效率指数（加权平均）
        const efficiencyIndex = (premiumShare * 0.5 + policyShare * 0.3 + (avgPremium / maxAvgPremium) * 100 * 0.2);

        return {
            name: channel,
            type: 'radar',
            data: [{
                value: [
                    premiumShare.toFixed(2),
                    policyShare.toFixed(2),
                    avgPremium.toFixed(2),
                    efficiencyIndex.toFixed(2)
                ],
                name: channel
            }]
        };
    });

    const option = {
        title: {
            text: '渠道效率对比分析',
            left: 'center'
        },
        tooltip: {
            trigger: 'item'
        },
        legend: {
            bottom: 0,
            data: channels
        },
        radar: {
            indicator: indicators,
            radius: '60%'
        },
        series: series
    };

    this.charts.channelEfficiency.setOption(option);
}

def analyze_weekday_weekend_comparison(self):
    """工作日vs周末业务对比分析"""
    print("【14/14】分析工作日vs周末对比...")

    result = {
        '对比统计': [],
        '险别分布': {}
    }

    if '是否周末' in self.df.columns and '签单保费' in self.df.columns:
        # 基础对比
        comparison = self.df.groupby('是否周末').agg({
            '签单保费': 'sum',
            '保单数': 'sum'
        }).reset_index()

        comparison['平均保费'] = comparison['签单保费'] / comparison['保单数']

        for _, row in comparison.iterrows():
            result['对比统计'].append({
                '类型': '周末' if row['是否周末'] == '是' else '工作日',
                '总保费': float(row['签单保费']),
                '保单数': int(row['保单数']),
                '平均保费': float(row['平均保费'])
            })

        # 险别分布对比
        if '险别' in self.df.columns:
            insurance_comparison = pd.crosstab(
                index=self.df['是否周末'],
                columns=self.df['险别'],
                values=self.df['签单保费'],
                aggfunc='sum'
            )

            for weekday_flag in insurance_comparison.index:
                label = '周末' if weekday_flag == '是' else '工作日'
                result['险别分布'][label] = {
                    ins: float(insurance_comparison.loc[weekday_flag, ins])
                    for ins in insurance_comparison.columns
                }

    self.analysis['工作日周末对比'] = result
    return result
```
