# 营业货车专项分析功能实现计划

## 📋 需求概述

在 `PremiumDashboard` 中添加"营业货车专项分析"标签页，包含：
- **玫瑰图**：各吨位分段的保费和保单数量（支持指标切换）
- **双Y分析图1**：按吨位分段的机构分析（左Y：保费柱状图，右Y：保费占比折线）
- **双Y分析图2**：按机构的吨位分段分析（左Y：保费柱状图，右Y：保费占比折线）
- **筛选器**：险类、新能源、过户、新车、续保（已有筛选器复用）

## 🎯 架构决策

1. **页面结构**：在 `PremiumDashboard.tsx` 中添加标签页切换（不是独立页面）
2. **指标切换**：玫瑰图内部状态管理保费/保单数量切换
3. **吨位分段**：直接从 Parquet 数据的 `tonnage_segment` 字段读取（已定义）
4. **占比计算**：相对于总保费的占比（使用 CTE + JOIN 模式）

```typescript
/**
 * 营业货车专项分析 SQL 生成器
 * 业务规则：
 * - 筛选条件：customer_category LIKE '%货车%'
 * - 吨位分段：tonnage_segment（可选字段，需处理NULL）
 * - 占比计算：使用CTE模式
 */

// 玫瑰图查询（支持保费和保单数两种指标）
export function generateTonnageRoseQuery(
  metric: 'premium' | 'count',
  whereClause: string = '1=1'
): string {
  const agg = metric === 'premium' ? 'SUM(premium)' : 'COUNT(DISTINCT policy_no)';
  return `
    SELECT
      COALESCE(tonnage_segment, '未知') as dim_key,
      ${agg} as value
    FROM PolicyFact
    WHERE ${whereClause}
      AND customer_category LIKE '%货车%'
    GROUP BY COALESCE(tonnage_segment, '未知')
    ORDER BY value DESC
  `;
}

// 双Y图1：按吨位分段的机构分析
export function generateOrgByTonnageQuery(whereClause: string = '1=1'): string {
  return `
    WITH tonnage_org_premium AS (
      SELECT
        COALESCE(tonnage_segment, '未知') as tonnage_segment,
        org_level_3,
        SUM(premium) as premium
      FROM PolicyFact
      WHERE ${whereClause}
        AND customer_category LIKE '%货车%'
      GROUP BY COALESCE(tonnage_segment, '未知'), org_level_3
    ),
    tonnage_totals AS (
      SELECT
        tonnage_segment,
        SUM(premium) as total_premium
      FROM tonnage_org_premium
      GROUP BY tonnage_segment
    )
    SELECT
      t.tonnage_segment,
      t.org_level_3,
      t.premium,
      (t.premium * 1.0 / NULLIF(tt.total_premium, 0)) as premium_ratio
    FROM tonnage_org_premium t
    JOIN tonnage_totals tt ON t.tonnage_segment = tt.tonnage_segment
    ORDER BY t.tonnage_segment, t.premium DESC
  `;
}

// 双Y图2：按机构的吨位分段分析
export function generateTonnageByOrgQuery(whereClause: string = '1=1'): string {
  return `
    WITH org_tonnage_premium AS (
      SELECT
        org_level_3,
        COALESCE(tonnage_segment, '未知') as tonnage_segment,
        SUM(premium) as premium
      FROM PolicyFact
      WHERE ${whereClause}
        AND customer_category LIKE '%货车%'
      GROUP BY org_level_3, COALESCE(tonnage_segment, '未知')
    ),
    org_totals AS (
      SELECT
        org_level_3,
        SUM(premium) as total_premium
      FROM org_tonnage_premium
      GROUP BY org_level_3
    )
    SELECT
      t.org_level_3,
      t.tonnage_segment,
      t.premium,
      (t.premium * 1.0 / NULLIF(ot.total_premium, 0)) as premium_ratio
    FROM org_tonnage_premium t
    JOIN org_totals ot ON t.org_level_3 = ot.org_level_3
    ORDER BY t.org_level_3, t.premium DESC
  `;
}
```

```typescript
import React, { useEffect, useRef } from 'react';
import * as echarts from 'echarts';

interface RoseChartProps {
  title: string;
  data: { name: string; value: number }[];
  loading?: boolean;
  height?: number;
  showValueLabel?: boolean;
}

export const RoseChart: React.FC<RoseChartProps> = ({
  title,
  data,
  loading = false,
  height = 400,
  showValueLabel = true,
}) => {
  const chartRef = useRef<HTMLDivElement>(null);
  const chartInstance = useRef<echarts.ECharts | null>(null);

  useEffect(() => {
    if (!chartRef.current) return;

    if (!chartInstance.current) {
      chartInstance.current = echarts.init(chartRef.current);
    }

    const chart = chartInstance.current;

    if (loading) {
      chart.showLoading();
      return;
    }

    chart.hideLoading();

    if (data.length === 0) {
      chart.setOption({
        title: { text: title, left: 'center' },
        graphic: {
          type: 'text',
          left: 'center',
          top: 'middle',
          style: { text: '暂无数据', fontSize: 16, fill: '#999' },
        },
      });
      return;
    }

    const option: echarts.EChartsOption = {
      title: {
        text: title,
        left: 'center',
        textStyle: { fontSize: 16, fontWeight: 'bold' },
      },
      tooltip: {
        trigger: 'item',
        formatter: '{b}: {c} ({d}%)',
      },
      legend: {
        orient: 'vertical',
        left: 'left',
        data: data.map(d => d.name),
      },
      series: [
        {
          name: title,
          type: 'pie',
          roseType: 'radius',  // 关键配置：玫瑰图模式
          radius: ['20%', '70%'],
          center: ['50%', '50%'],
          data: data,
          label: {
            show: showValueLabel,
            formatter: '{b}\n{d}%',
          },
          itemStyle: {
            borderRadius: 5,
          },
        },
      ],
    };

    chart.setOption(option);

    // Resize handler
    const handleResize = () => chart.resize();
    window.addEventListener('resize', handleResize);

    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, [title, data, loading, height, showValueLabel]);

  return <div ref={chartRef} style={{ height: `${height}px` }} />;
};
```

#### 3.1 创建 `src/features/dashboard/TruckAnalysisPanel.tsx`

```typescript
import React, { useEffect, useCallback } from 'react';
import { duckdbClient } from '../../shared/duckdb/client';
import { buildWhereClauseFromFilters } from '../../shared/utils/queryBuilder';
import {
  generateTonnageRoseQuery,
  generateOrgByTonnageQuery,
  generateTonnageByOrgQuery,
} from '../../shared/sql/truck';
import { TonnageRoseChart } from '../../widgets/charts/TonnageRoseChart';
import { OrgByTonnageDualYChart } from '../../widgets/charts/OrgByTonnageDualYChart';
import { TonnageByOrgDualYChart } from '../../widgets/charts/TonnageByOrgDualYChart';
import { AdvancedFilterState } from '../filters/AdvancedFilterPanel';

interface TruckAnalysisPanelProps {
  filters: AdvancedFilterState;
}

export const TruckAnalysisPanel: React.FC<TruckAnalysisPanelProps> = ({ filters }) => {
  const [rosePremiumData, setRosePremiumData] = useState([]);
  const [roseCountData, setRoseCountData] = useState([]);
  const [orgByTonnageData, setOrgByTonnageData] = useState([]);
  const [tonnageByOrgData, setTonnageByOrgData] = useState([]);
  const [loading, setLoading] = useState(false);

  const refreshData = useCallback(async () => {
    setLoading(true);
    try {
      const whereClause = buildWhereClauseFromFilters(filters);
      const batchId = duckdbClient.startBatch();

      // 并行查询3个图表的数据
      const [premiumTable, countTable, orgByTonnageTable, tonnageByOrgTable] =
        await Promise.all([
          duckdbClient.query(generateTonnageRoseQuery('premium', whereClause)).promise,
          duckdbClient.query(generateTonnageRoseQuery('count', whereClause)).promise,
          duckdbClient.query(generateOrgByTonnageQuery(whereClause)).promise,
          duckdbClient.query(generateTonnageByOrgQuery(whereClause)).promise,
        ]);

      if (!duckdbClient.isBatchValid(batchId)) return;

      // Arrow Table → ECharts 格式转换
      setRosePremiumData(premiumTable.toArray().map(row => ({
        name: row.dim_key,
        value: row.value,
      })));

      setRoseCountData(countTable.toArray().map(row => ({
        name: row.dim_key,
        value: row.value,
      })));

      setOrgByTonnageData(orgByTonnageTable.toArray());
      setTonnageByOrgData(tonnageByOrgTable.toArray());
    } catch (err) {
      console.error('Failed to load truck analysis data:', err);
    } finally {
      setLoading(false);
    }
  }, [filters]);

  useEffect(() => {
    refreshData();
  }, [refreshData]);

  return (
    <div className="space-y-6">
      {/* 玫瑰图 */}
      <TonnageRoseChart
        premiumData={rosePremiumData}
        countData={roseCountData}
        loading={loading}
      />

      {/* 双Y图1 */}
      <OrgByTonnageDualYChart
        data={orgByTonnageData}
        loading={loading}
      />

      {/* 双Y图2 */}
      <TonnageByOrgDualYChart
        data={tonnageByOrgData}
        loading={loading}
      />
    </div>
  );
};
```

## 🚨 技术风险与缓解措施

### 风险1：`tonnage_segment` 字段缺失
**风险等级**：🟡 中
**缓解措施**：
- SQL中使用 `COALESCE(tonnage_segment, '未知')` 处理NULL值
- 在UI中说明："如果数据中缺少吨位分段，所有记录归入'未知'分组"

### 风险2：数据中无货车记录
**风险等级**：🟢 低
**缓解措施**：
- 图表组件已有"暂无数据"提示
- 筛选条件：`customer_category LIKE '%货车%'` 自动过滤

### 风险3：占比计算除零错误
**风险等级**：🟢 低
**缓解措施**：
- SQL中使用 `NULLIF(total_premium, 0)` 防止除零
- 前端图表组件处理 `null` 值

### 风险4：性能问题（数据量大）
**风险等级**：🟢 低
**缓解措施**：
- 使用 `startBatch()` 批量查询（避免多次序列化）
- 使用 CTE 避免重复扫描
- 前端已有虚拟滚动组件（VirtualTable）
